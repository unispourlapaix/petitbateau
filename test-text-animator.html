📝 Exemples d'utilisation

// Typewriter avec callback
const anim1 = new TextAnimator('demo1');
anim1.typewriter('Hello Emmanuel !', 50, () => {
  console.log('Animation terminée!');
});
// Animation digital avec couleurs
const anim2 = new TextAnimator('demo3');
anim2.setColors('#00cc44', '#64ffda')
     .digital('Texte avec couleurs personnalisées', 40);
// Support HTML
const anim3 = new TextAnimator('demo3');
anim3.typewriter('<strong>Texte en gras</strong>\nAvec saut de ligne');
